JavaScript Scope

Scope

is the accessibility of variables, functions, and objects in some particular part of your code during runtime

Types of Scope

  • Global
  • Local

Global scope


var global = 'Global variable.';
function saySomething() {
  console.log(global);
}
console.log(global); // 'Global variable.'
saySomething(); // 'Global variable.'

Declaration with var


var variable = 'Variable';
var variable = 'Other Variable';
console.log(variable); //'Other Variable'

Block Statements


if (true) {
    // this 'if' conditional block doesn't create a new scope
    var name = 'Hammad'; // name is still in the global scope
}
console.log(name); // logs 'Hammad'

Declaration with let/const


var name = 'Hammad';
let likes = 'Coding';
const skills = 'Javascript and PHP';

Automatically Global


myFunction();
// code here can use carName
function myFunction() {
  carName = "Volvo";
}

Global Variables in HTML

global = window


var carName = "Volvo";
console.log(carName); //"Volvo"
console.log(window.carName); //"Volvo"

Local Scope

Variables are usable only in a specific part of the code

  • function scope
  • block scope(ES2015+)

Function scope

Variables defined inside a function are in the local scope


// Global Scope
function someFunction() {
    // Local Scope #1
}
// Global Scope
function anotherFunction() {
    // Local Scope #2
}
// Global Scope

Usage the same variable name in different functions


function firstFunction(){
  var carName = 'Volvo';
  console.log(carName); // Volvo
}

function secondFunction(){
  var carName = 'Volvo';
  console.log(carName); // Volvo
}

Functions do not have access to each other's scopes


function firstFunction(){
  var numberOfFunction = 1;
  var carName = 'Volvo';
  console.log(carName); // Volvo
}

function secondFunction(){
  var carName = 'Audi';
  console.log(carName); // Audi
  firstFunction();
  console.log(numberOfFunction); //Error
}

Block scope

Variable hello is scoped to the curly brace


{
  const hello = 'Hello World!'
  console.log(hello) // 'Hello World!'
}

console.log(hello) // Error, hello is not defined

Nested scopes

the ability of an inner function to access the scope of an outer function


function outerFunction () {
  const outer = `Outer function`

  function innerFunction() {
    const inner = `Inner function`
    console.log(outer) // Outer function
  }

  console.log(inner) // Error, inner is not defined
}

Scope shadowing

the priority of the local variable over the global


var scope = "global"; //Declare a global variable
function checkscope() {
  var scope = "local"; //Declare a local variable with the same name
  return scope; //Return the local value, not the global one
}
checkscope() //"local"
